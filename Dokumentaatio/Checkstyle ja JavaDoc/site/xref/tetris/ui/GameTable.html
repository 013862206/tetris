1   package tetris.ui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import tetris.entity.*;
7   import tetris.logic.*;
8   
9   /**
10   *GameTable piirtää pelilaudan tilanteen.
11   */
12  public class GameTable extends JPanel {
13  
14      private Game game;
15      private int scale;
16  
17      /**
18       *
19       * @param game
20       * @param scale
21       */
22      public GameTable(Game game, int scale) {
23          this.game = game;
24          this.scale = scale;
25      }
26  
27      @Override
28      public void paintComponent(Graphics g) {
29          super.paintComponent(g);
30          Part part;
31          for (int y = 0; y < game.getTable().getHeight(); y++) {
32              for (int x = 0; x < game.getTable().getWidth(); x++) {
33                  if (game.getTable().getTable()[y][x] == null) {
34                      g.setColor(Color.white);
35                      g.drawRect(scale * x, scale * y, scale, scale);
36                  } else {
37                      part = game.getTable().getTable()[y][x];
38                      g.setColor(part.getColor());
39                      g.fill3DRect(scale * x, scale * y, scale, scale, true);
40                  }
41              }
42          }
43          Block block = game.getCurrentBlock();
44          for (Part p : block.getParts()) {
45              g.setColor(p.getColor());
46              g.fill3DRect(scale * p.getXCoordinate(), scale * p.getYCoordinate(), scale, scale, true);
47          }
48      }
49  }
