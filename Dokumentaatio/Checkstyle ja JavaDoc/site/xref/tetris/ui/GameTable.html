23  /**
24   * GameTable piirtää pelilaudan tilannetta ja sisältää peliloopin.
25   */
26  public class GameTable extends JPanel implements ActionListener {
27  
28      private Game game;
29      private int scale;
30      private JFrame frame;
31      private Timer timer;
32      private Menu menu;
33      private boolean pause;
34  
35      /**
36       * Luo uuden GameTable olion, joka piirtää sille parametrina annettavaa
37       * peliä. GameTable tuntee myös pelivalikon, josta GameTable on luotu, jotta
38       * pelivalikkoon voidaan myöhemmin palata.
39       *
40       * @param game Peli, jota GameTable piirtää
41       * @param scale Skaalauskerroin, joka määrittelee piirrettävien palojen koon
42       * @param menu Pelivalikko, joka luo GameTable olion
43       */
44      public GameTable(Game game, int scale, Menu menu) {
45          this.menu = menu;
46          this.game = game;
47          this.scale = scale;
48          frame = new JFrame("Tetris");
49          initialize();
50          pause = false;
51          timer = new Timer(1000 / game.getGameLevel(), this);
52          timer.start();
53      }
54  
55      @Override
56      public void paintComponent(Graphics g) {
57          super.paintComponent(g);
58          drawTable(g);
59          drawCurrentBlock(g);
60          drawPointStatistics(g);
61          drawNextBlock(g);
62          if (!game.getIsOn()) {
63              drawGameOver(g);
64          }
65      }
66  
67      private void drawTable(Graphics g) {
68          Part part;
69          for (int y = 0; y < game.getTable().getHeight(); y++) {
70              for (int x = 0; x < game.getTable().getWidth(); x++) {
71                  if (game.getTable().getTable()[y][x] == null) {
72                      g.setColor(Color.white);
73                      g.drawRect(scale * x, scale * y, scale, scale);
74                  } else {
75                      part = game.getTable().getTable()[y][x];
76                      g.setColor(part.getColor());
77                      g.fill3DRect(scale * x, scale * y, scale, scale, true);
78                  }
79              }
80          }
81          g.setColor(Color.gray);
82          g.drawRect(0, 0, game.getTable().getWidth() * scale, game.getTable().getHeight() * scale);
83      }
84  
85      private void drawCurrentBlock(Graphics g) {
86          Block block = game.getCurrentBlock();
87          for (Part p : block.getParts()) {
88              g.setColor(p.getColor());
89              g.fill3DRect(scale * p.getXCoordinate(), scale * p.getYCoordinate(), scale, scale, true);
90          }
91      }
92  
93      private void drawPointStatistics(Graphics g) {
94          Font font = new Font("Helvetica", Font.BOLD, 30);
95          g.setFont(font);
96          g.setColor(Color.black);
97          g.drawString("Game Level: " + game.getPointStatistics().getGameLevel(),
98                  game.getTable().getWidth() * scale + 50, 50);
99          g.drawString("Points: " + game.getPointStatistics().getPoints(),
100                 game.getTable().getWidth() * scale + 50, 100);
101         g.drawString("Blocks: " + game.getPointStatistics().getBlocks(),
102                 game.getTable().getWidth() * scale + 50, 150);
103         g.drawString("Lines: " + game.getPointStatistics().getLines(),
104                 game.getTable().getWidth() * scale + 50, 200);
105     }
106 
107     private void drawNextBlock(Graphics g) {
108         Block block = game.getNextBlock();
109         int x = block.getX();
110         int y = block.getY();
111         block.move(-x, -y);
112         for (Part p : block.getParts()) {
113             g.setColor(p.getColor());
114             g.fill3DRect(scale * (game.getTable().getWidth() + p.getXCoordinate() + 1) + 50,
115                     250 + scale * p.getYCoordinate(), scale, scale, true);
116         }
117         block.move(x, y);
118     }
119 
120     private void drawGameOver(Graphics g) {
121         Font font = new Font("Helvetica", Font.BOLD, 50);
122         g.setFont(font);
123         g.setColor(Color.red);
124         g.drawString("GAME OVER!", scale * 3, scale * 10);
125     }
126 
127     private void initialize() {
128         int width = (game.getTable().getWidth()) * scale + 350;
129         int height = (game.getTable().getHeight() + 2) * scale;
130 
131         frame.setPreferredSize(new Dimension(width, height));
132 
133         frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
134 
135         createComponents(frame.getContentPane());
136 
137         frame.pack();
138         frame.setLocationRelativeTo(null);
139         frame.setVisible(true);
140     }
141 
142     private void createComponents(Container container) {
143         JPanel buttonsPanel = new JPanel();
144         JButton menu = new JButton("MENU");
145         JButton exit = new JButton("EXIT");
146         JButton newGame = new JButton("NEW GAME");
147         JButton pause = new JButton("PAUSE");
148         menu.setFocusable(false);
149         exit.setFocusable(false);
150         newGame.setFocusable(false);
151         pause.setFocusable(false);
152 
153         menu.addActionListener(this);
154         exit.addActionListener(this);
155         newGame.addActionListener(this);
156         pause.addActionListener(this);
157 
158         GridLayout gl = new GridLayout();
159         gl.setColumns(4);
160         buttonsPanel.setLayout(gl);
161         buttonsPanel.add(newGame);
162         buttonsPanel.add(menu);
163         buttonsPanel.add(exit);
164         container.add(buttonsPanel, BorderLayout.NORTH);
165         container.add(this, BorderLayout.CENTER);
166 
167         GameKeyListener listener = new GameKeyListener(game, this);
168         frame.addKeyListener(listener);
169     }
170 
171     @Override
172     public void actionPerformed(ActionEvent e) {
173         timer.setDelay(1000 / game.getGameLevel());
174         if (!game.getIsOn()) {
175             timer.stop();
176         }
177         game.moveBlockDown();
178         if ("EXIT".equals(e.getActionCommand())) {
179             System.exit(0);
180         }
181         if ("MENU".equals(e.getActionCommand())) {
182             menu.getFrame().setVisible(true);
183             frame.dispose();
184         }
185         if ("NEW GAME".equals(e.getActionCommand())) {
186             game = new Game(menu.getGameLevel(), game.getTable().getWidth(), game.getTable().getHeight());
187             initialize();
188             timer.restart();
189         }
190         repaint();
191     }
192 }
