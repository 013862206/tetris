1   package tetris.logic;
2   
3   import java.awt.Color;
4   import java.util.Random;
5   import tetris.entity.*;
6   
7   /**
8    * Uusien pelipalikoiden arvonnan suorittava luokka.
9    */
10  public class BlockRandomizer {
11  
12      private Random random;
13      private int x;
14      private int y;
15  
16      /**
17       * Luo uuden pelipalikoiden arpojan.
18       *
19       * @param x Konstruoitavien palikoiden x-koordinaatti
20       * @param y Konstruoitavien palikoiden y-koordinaatti
21       * @param random Arvonnan suorittava arpoja
22       */
23      public BlockRandomizer(int x, int y, Random random) {
24          this.random = random;
25          this.x = x;
26          this.y = y;
27      }
28  
29      /**
30       * Arpoo uuden pelipalikan kaikkien mahdollisten palikoiden joukosta.
31       *
32       * @return Arvottu pelipalikka
33       */
34      public Block drawNewBlock() {
35          int a = random.nextInt(7);
36          switch (a) {
37              case 0:
38                  return new IBlock(x, y, Color.cyan);
39              case 1:
40                  return new JBlock(x, y, Color.blue);
41              case 2:
42                  return new LBlock(x, y, Color.orange);
43              case 3:
44                  return new OBlock(x, y, Color.yellow);
45              case 4:
46                  return new SBlock(x, y, Color.green);
47              case 5:
48                  return new TBlock(x, y, Color.magenta);
49              default:
50                  return new ZBlock(x, y, Color.red);
51          }
52      }
53  
54      /**
55       * @return Uusien palikoiden x-koordinaatti
56       */
57      public int getX() {
58          return x;
59      }
60  
61      /**
62       * @return Uusien palikoiden y-koordinaatti
63       */
64      public int getY() {
65          return y;
66      }
67  
68      /**
69       * @return Arvonnan suorittava arpoja
70       */
71      public Random getRandom() {
72          return random;
73      }
74  }
